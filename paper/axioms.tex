\caption{
Blue lines show the functions $y=x$, $y=2x$, $y=3x$, and $y=4x$, which provide lower and upper linear bounds
for $x^2$ on unit intervals between $0$ and $4$. For each $x$, the active bounds are the maximal lower bound
and the minimal upper bound among all available bounds (shown as solid lines).}
\end{figure}

\begin{example}

\begin{align*}
	x^3 + y^3 + z^3 = 79.
\end{align*}
\end{example}
