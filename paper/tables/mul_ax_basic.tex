		\hline
		\textit{Eq-zero} & $v = 0$                            & $x = 0 \implies \pure{x^k y^l} = 0$                   \\

		\hline
		\textit{Eq}      & $v \neq 0$,\ $\pure{y^l}$ exists   & $x = v \implies \pure{x^k y^l} = v^k \pure{y^l}$      \\
		                 & $w \neq 0$,\ $\pure{x^k}$ exists   & $y = w \implies \pure{x^k y^l} = w^l \pure{x^k}$      \\
		                 & otherwise                          & $x = v \land y = w \implies \pure{x^k y^l} = v^k w^l$ \\
		\hline
		\textit{Mod}     & $v \neq 0,\ w \neq 0$              & $x \equiv v \pmod{m} \land y \equiv w \pmod{m}$       \\
		                 &                                    & $\implies \pure{x^k y^l} \equiv v^k w^l \pmod{m}$     \\
		\hline
